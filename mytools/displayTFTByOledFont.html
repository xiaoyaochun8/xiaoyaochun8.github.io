使用OLED 16×16 字模数据 模拟显示TFT屏幕

// ========== OLED 16×16 字模 专用绘制代码 ========== for (let i = 0; i < 16; i++) { // 行 0~15 for (let j = 0; j < 16; j++) { // 列 0~15 // ===== OLED 字模 核心计算（已改好）===== const byteIdx = j + (i >> 3) * 16; // 计算字节索引 const bit = 1 << (i & 0x07); // 计算位掩码 if (c.dot[byteIdx] & bit) { RBTFT18.drawPixel(x + j, y + i, Color.Red); } else { RBTFT18.drawPixel(x + j, y + i, Color.Black); } } }
